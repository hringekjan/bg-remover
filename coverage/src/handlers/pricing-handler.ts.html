/**

    statusCode,
    headers: {
      'Content-Type': 'application/json',
      'X-Cache-Hit-Rate': ((body.cacheMetrics?.hitRate || 0) * 100).toFixed(1),

      'Access-Control-Allow-Methods': 'POST, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type, x-tenant-id',

}
